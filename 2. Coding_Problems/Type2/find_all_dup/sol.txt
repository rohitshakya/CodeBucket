
class Solution {
public:
    vector<int> findDuplicates(vector<int>& nums) {
 vector<int> res;
        for(int i=0;i<nums.size();i++)
        {
            int pos=abs(nums[i])-1;
            nums[pos]=-nums[pos];
        
            if (nums[pos] > 0)    // -ve * -ve = +ve
                res.push_back(pos+1);
        }
       return res; 
    }
};
