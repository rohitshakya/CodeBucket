
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int curr_max=nums[0];
        int maximum=nums[0];
        
        for(int i=1;i<nums.size();i++)
        {
            curr_max= max( nums[i] , nums[i]+curr_max);
            maximum= max( curr_max , maximum );
        }
        return maximum;
    }
};
